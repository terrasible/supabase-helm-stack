🔧 POST-INSTALLATION STEPS:

1. Wait for all pods to be ready:
   kubectl get pods -n {{ .Release.Namespace }} -w

2. Check dependency chart status:
   helm list -n {{ .Release.Namespace }}

{{- if .Values.postgresql.enabled }}
3. Get PostgreSQL credentials:
   export POSTGRES_PASSWORD=$(kubectl get secret --namespace {{ .Release.Namespace }} {{ .Release.Name }}-postgresql -o jsonpath="{.data.postgres-password}" | base64 -d)
   echo "PostgreSQL Password: $POSTGRES_PASSWORD"
{{- end }}

4. Port-forward to access Studio locally:
   kubectl port-forward -n {{ .Release.Namespace }} svc/{{ include "supabase.studio.fullname" . }} 3000:3000
   Studio URL: http://localhost:3000

5. Access API Gateway:
   {{- if .Values.kongOperator.enabled }}
   kubectl port-forward -n {{ .Values.kongOperator.serviceNamespace }} svc/{{ .Values.kongOperator.serviceName }} 8000:{{ .Values.kongOperator.servicePort }}
   {{- else }}
   # Kong Operator not enabled - configure Kong Gateway separately
   {{- end }}
   API URL: http://localhost:8000/rest/v1
